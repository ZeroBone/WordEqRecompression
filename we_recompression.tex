\begin{frame}
\frametitle{The problem}

Fix some finite alphabet $ \Gamma $ and set of variables $ \X $.

\begin{problem}[Satisfiability of word equations]
	Given an equation $ U = V $ where $ U, V \in (\Gamma \cup \X)^* $, does there exist a substitution $ \varphi : \X \rightarrow \Gamma^* $ satisfying the equation?
\end{problem}

\begin{example}[Satisfiable equation]
	The equation $ b\var{x}ba\var{y}b = \var{y}bb\var{x}ab $ is satisfiable via $ \varphi(\var{x}) := \sol{ab} $, $ \varphi(\var{y}) := \sol{ba} $: \begin{center}
		\begin{tabular}{c c c c c c c c c c}
			$ b $ & $ \var{x} $ & $ ba $ & $ \var{y} $ & $ b $ & $ = $ & $ \var{y} $ & $ bb $ & $ \var{x} $ & $ ab $ \\
			$ b $ & $ \sol{ab} $ & $ ba $ & $ \sol{ba} $ & $ b $ & $ = $ & $ \sol{ba} $ & $ bb $ & $ \sol{ab} $ & $ ab $
		\end{tabular}
	\end{center}
\end{example}

\begin{example}[Unsatisfiable equations]
	The equations $ a\var{x} = b\var{y} $, $ a\var{x}\var{y} = \var{y}a\var{x}a $ and $ \var{x}a = ab\var{x} $ are unsatisfiable.
\end{example}

\end{frame}

\begin{frame}
\frametitle{History}

\begin{itemize}
	\item[1977] Word equations are decidable \cite{makanin:1977}; very high complexity;
	\item[1990] Makanin's algorithm \cite{makanin:1977} improved to run in $ \FOUREXP $ \cite{jaffar:1990, schulz:1990};
	\item[1996] Further improvement to $ \THREEEXP $ \cite{koscielski_pacholski:1996};
	\item[1998] Further improvement to $ \EXPSPACE $ \cite{gutierrez:1998};
	\item[1999] $ \NEXP $ algorithm via a double-exponential upper bound on the size of the length-minimal solution \cite{plandowski:1999};
	\item[2004] $ \PSPACE $ algorithm via rewriting system \cite{plandowski:2004}. Soundness trivial, complicated proof of completeness;
	\item[2016] $ \PSPACE $ algorithm via the \textit{recompression technique} \cite{jez:2016};
\end{itemize}

\begin{openproblem}{Long-standing open problem}
	Are word equations in $ \NP $?
\end{openproblem}
